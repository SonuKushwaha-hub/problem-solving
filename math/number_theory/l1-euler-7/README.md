(define mark-primes
  (lambda (d n v)
    (cond
      [(<= d n)
       (cond
         [(= 0 (vector-ref v d))
          (let loop [(i (* d d))]
            (cond
              [(<= i n)
               (vector-set! v i 1)
               (loop (+ i d))]))])
       (mark-primes (add1 d) n v)])))

(define nth-prime
  (lambda (n v bsz)
    (vector-set! v 0 1)
    (vector-set! v 1 1)
    (mark-primes 2 bsz v)
    (let loop [(c 0)
               (i 2)]
      (cond
        [(= 0 (vector-ref v i))
         ;(writeln i)
         (cond
           [(= (add1 c) n)
            i]
           [#t
            (loop (add1 c) (add1 i))])]
        [#t
         (loop c (add1 i))]))))
